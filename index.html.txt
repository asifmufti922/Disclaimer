                });

            // SEO Analysis function
            function runSEOAnalysis(ownerName, siteName, website, disclaimerHTML) {
                let score = 0;
                let recommendations = [];
                
                // Check for required elements
                if(ownerName && ownerName !== 'Your Name') score += 15;
                if(siteName && siteName !== 'YourSiteName') score += 15;
                if(website && website !== 'https://yourwebsite.com') score += 15;
                
                // Check content length
                const textContent = disclaimerHTML.replace(/<[^>]*>/g, '');
                if(textContent.length > 1000) score += 15;
                if(textContent.length > 1500) score += 10;
                
                // Check for keywords
                const keywords = ['disclaimer', 'liability', 'legal', 'terms', 'affiliate'];
                let keywordCount = 0;
                keywords.forEach(keyword => {
                    if(textContent.toLowerCase().includes(keyword)) keywordCount++;
                });
                score += (keywordCount / keywords.length) * 20;
                
                // Check for structured content
                if(disclaimerHTML.includes('<h2') && disclaimerHTML.includes('</h2>')) score += 10;
                
                // Generate recommendations
                if(ownerName === 'Your Name') {
                    recommendations.push('Add your real name or company name for better credibility');
                }
                
                if(siteName === 'YourSiteName') {
                    recommendations.push('Customize your website name for a professional look');
                }
                
                if(website === 'https://yourwebsite.com') {
                    recommendations.push('Add your actual website URL for proper attribution');
                }
                
                if(textContent.length < 1000) {
                    recommendations.push('Consider adding more detailed legal information (currently ' + textContent.length + ' characters)');
                }
                
                if(keywordCount < 3) {
                    recommendations.push('Add more legal keywords for better SEO (found ' + keywordCount + ' of 5 recommended keywords)');
                }
                
                // Display results
                const seoResults = document.getElementById('seoResults');
                const seoScore = document.getElementById('seoScore');
                const scoreValue = document.getElementById('scoreValue');
                const seoProgress = document.getElementById('seoProgress');
                
                seoScore.style.display = 'block';
                scoreValue.textContent = Math.min(Math.round(score), 100);
                seoProgress.style.width = Math.min(Math.round(score), 100) + '%';
                
                let resultsHTML = '<h4 style="color: ' + (score >= 70 ? '#28a745' : score >= 40 ? '#ffc107' : '#dc3545') + ';">SEO Recommendations</h4>';
                
                if(recommendations.length > 0) {
                    resultsHTML += '<ul style="margin-top: 1rem; padding-left: 1.5rem;">';
                    recommendations.forEach(rec => {
                        resultsHTML += '<li style="margin-bottom: 0.5rem;">' + rec + '</li>';
                    });
                    resultsHTML += '</ul>';
                } else {
                    resultsHTML += '<p style="color: #28a745;"><i class="fas fa-check-circle"></i> Great job! Your disclaimer meets all basic SEO requirements.</p>';
                }
                
                // Add positive points
                resultsHTML += '<h4 style="margin-top: 1.5rem; color: #28a745;">What\'s Working Well</h4>';
                resultsHTML += '<ul style="margin-top: 1rem; padding-left: 1.5rem;">';
                
                if(ownerName && ownerName !== 'Your Name') {
                    resultsHTML += '<li style="margin-bottom: 0.5rem;">Owner name is properly set</li>';
                }
                
                if(siteName && siteName !== 'YourSiteName') {
                    resultsHTML += '<li style="margin-bottom: 0.5rem;">Website name is properly set</li>';
                }
                
                if(website && website !== 'https://yourwebsite.com') {
                    resultsHTML += '<li style="margin-bottom: 0.5rem;">Website URL is properly set</li>';
                }
                
                if(textContent.length > 1000) {
                    resultsHTML += '<li style="margin-bottom: 0.5rem;">Detailed content (' + textContent.length + ' characters)</li>';

                    resultsHTML += '<li style="margin-bottom: 0.5rem;">Good keyword coverage (' + keywordCount + ' of 5)</li>';
                }
                
                resultsHTML += '</ul>';
                
                seoResults.innerHTML = resultsHTML;

                document.querySelector('.tab[data-tab="seo"]').click();
            }
